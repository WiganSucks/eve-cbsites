function parsePaste(paste) { const rows = paste.trim().split('\n'); const results = []; for (const row of rows) { const cols = row.split('|'); if (cols.length < 4) continue; // Make sure there are enough columns const scanType = cols[1].trim(); // Cosmic Signature/Anomaly const siteType = cols[2].trim(); // Combat Site etc const siteName = cols[3].trim(); if (siteType === 'Combat Site') { // Filtering for Combat Site results.push({ scanType, siteType, siteName }); } } return results; } function renderResults(results) { const output = results.map(result => `Scan Type: ${result.scanType}, Site Type: ${result.siteType}, Site Name: ${result.siteName}`).join('\n'); return output; } // Update placeholder: // Assume there's a function responsible for setting placeholders, replacing contents with appropriate values updatePlaceholderText('Scan Type | Site Type | Site Name');
